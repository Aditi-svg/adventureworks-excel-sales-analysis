-- Totals
[Total Sales]           = SUM(Sales[Sales Amount])
[Total Extended Amount] = SUM(Sales[Extended Amount])
[Total Product Cost]    = SUM(Sales[Total Product Cost])
[Units Sold]            = SUM(Sales[Order Quantity])

-- Profitability
[Profit]          = [Total Sales] - [Total Product Cost]
[Gross Margin %]  = DIVIDE([Profit], [Total Sales])

-- Orders
[Distinct Orders] = DISTINCTCOUNT('Sales order'[Sales Order])
[Order Lines]     = DISTINCTCOUNT('Sales order'[Sales Order Line])

-- AOV
[AOV] = DIVIDE([Total Sales], [Distinct Orders])

-- Discounting
[Discount Amount] = [Total Extended Amount] - [Total Sales]
[Discount %]      = DIVIDE([Discount Amount], [Total Extended Amount])

-- Time intelligence (requires 'Date'[Date] marked as Date Table)
[Sales MTD] = TOTALMTD([Total Sales], 'Date'[Date])
[Sales QTD] = TOTALQTD([Total Sales], 'Date'[Date])
[Sales YTD] = TOTALYTD([Total Sales], 'Date'[Date])
[Sales LY]  = CALCULATE([Total Sales], SAMEPERIODLASTYEAR('Date'[Date]))
[YoY %]     = DIVIDE([Total Sales] - [Sales LY], [Sales LY])

-- Rolling
[Rolling 12M Sales] =
CALCULATE([Total Sales], DATESINPERIOD('Date'[Date], MAX('Date'[Date]), -12, MONTH))
